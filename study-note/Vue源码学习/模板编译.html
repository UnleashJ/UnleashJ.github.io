2022/4/3 模板编译解析器优化器代码生成器

# 概念

模板编译的主要目标是生成渲染函数，渲染函数的作用是每次执行它就会使用当前最新的状态生成vnode，然后使用vnode进行渲染。

# 模板编译成渲染函数的步骤

由于静态节点不需要总是重新渲染，所以可以将模板编译成渲染函数分为三个步骤：首先将模板解析成抽象语法树AST，然后遍历AST给静态节点做一个标记（这样在虚拟DOM更新节点时，发现该标记就不会重新渲染它），最后使用AST生成渲染函数。这三部分内容在模板编译中抽象出三个模块实现各自功能，分别是：解析器，优化器，代码生成器。

# 解析器

解析器是为了将模板解析成抽象语法树AST。AST与vnode有点类似，都是使用JavaScript对象表示节点。很多个独立的节点通过parent属性和children属性连在一起时，就变成了一棵树。

在解析器内部生成了很多小解析器，包括过滤器解析器，文本解析器，HTML解析器等。文本解析器主要用来解析带变量的文本，不带变量的文本不需要解析。HTML解析器在解析HTML的过程会不断触发各种钩子函数，包括：开始start标签钩子函数、end结束标签钩子函数、chars文本钩子函数、comment注释钩子函数。

# HTML解析器

HTML解析器在解析HTML的过程会不断触发各种钩子函数，包括：开始start标签钩子函数、end结束标签钩子函数、chars文本钩子函数、comment注释钩子函数。我们可以在钩子函数中构建AST节点，在start钩子函数中构建元素类型的节点，在chars中构建文本类型的节点，在comment中构建注释类型的节点。start钩子函数有三个参数：tag（标签名）、attrs（标签的属性）、unary（是否自闭合标签）。

parseHTML(template,{
	start(tag,attrs,unary){
        //解析到标签的开始位置，触发该函数，可以构建元素类型的AST节点
    },
    end(){
        //解析到标签结束的位置，触发该函数
    },
    chars(text){
        //解析文本时，触发该函数，可以构建文本类型的AST节点
    },
    comment(text){
        //解析注释时，触发该函数，可以构建注释类型的AST节点
    }
})

解析HTML时，是按照顺序从上往下从左往右，所以构建AST层级关系可以使用栈（先进后出）。在每次触发start钩子函数时，把当前构建的节点推入栈中，触发end钩子函数时，就从栈中弹出一个节点。栈的最后一个节点是当前构建节点的父节点。

# 截取文本

如果HTML模板的第一个字符不是<，那么他就是文本。只需要找到下一个‘<’在什么地方，这之间的字符都是文本。

while(html){
    let text = ''
    let textEndIndex = html.indexOf('<')
    if(textEndIndex>=0){
        text = html.substring(0,textEndIndex)
        html = html.substring(textEndIndex)
    }
    if(textEndIndex<0){
        text = html//整段都是文本

    if(options.chars && text){
        options.chars(text)

截取文本时，若<是文本内容的一部分，例如1<2</div>此时上面的方法只能获取到1。解决方法：在截取完<前面的字符之后，若剩下的模板不符合任何片段（开始变迁、结束变迁、注释、条件注释等）类型，就说明这个<是文本内容的一部分。

while(html){
    let text,rest,next
    let textEndIndex = html.indexOf('<')
    if(textEndIndex>=0){
        rest = html.slice(textEndIndex);//‘<’(包含)之后的html内容
        while(
        	!endTag.test(rest)&&//剩余的不是结束标签

        text = html//整段都是文本
        html= ''
    }
    if(options.chars && text){
        options.chars(text)
    }
}

# 纯文本内容元素的处理

script、style、textarea三种元素叫作纯文本内容元素。如果当前截取元素的父元素是纯文本内容元素，那么只需要把这些内容当做文本截取，并触发钩子函数chars，然后再将结束标签截取出来触发钩子函数end。也就是说如果父标签是纯文本内容元素，那么本轮循环会一次性将这个父标签给处理完毕。

# 整体逻辑实现（Vue.js源码）

function parseHTML (html, options) {
    var stack = [];
    var expectHTML = options.expectHTML;
    var isUnaryTag$$1 = options.isUnaryTag || no;
    var canBeLeftOpenTag$$1 = options.canBeLeftOpenTag || no;
    var index = 0;
    var last, lastTag;
    while (html) {
      last = html;
      // Make sure we're not in a plaintext content element like script/style
      if (!lastTag || !isPlainTextElement(lastTag)) {
        var textEnd = html.indexOf('<');
        if (textEnd === 0) {
          // Comment:
          if (comment.test(html)) {
            var commentEnd = html.indexOf('-->');

            if (commentEnd >= 0) {
              if (options.shouldKeepComment) {
                options.comment(html.substring(4, commentEnd), index, index + commentEnd + 3);
              }
              advance(commentEnd + 3);
              continue
            }
          }

          // http://en.wikipedia.org/wiki/Conditional_comment#Downlevel-revealed_conditional_comment
          if (conditionalComment.test(html)) {
            var conditionalEnd = html.indexOf(']>');

            if (conditionalEnd >= 0) {
              advance(conditionalEnd + 2);
              continue
            }
          }

          // Doctype:
          var doctypeMatch = html.match(doctype);
          if (doctypeMatch) {
            advance(doctypeMatch[0].length);
            continue
          }

          // End tag:
          var endTagMatch = html.match(endTag);
          if (endTagMatch) {
            var curIndex = index;
            advance(endTagMatch[0].length);
            parseEndTag(endTagMatch[1], curIndex, index);
            continue
          }

          // Start tag:
          var startTagMatch = parseStartTag();
          if (startTagMatch) {
            handleStartTag(startTagMatch);
            if (shouldIgnoreFirstNewline(startTagMatch.tagName, html)) {
              advance(1);
            }
            continue
          }
        }

        var text = (void 0), rest = (void 0), next = (void 0);
        if (textEnd >= 0) {
          rest = html.slice(textEnd);
          while (
            !endTag.test(rest) &&
            !startTagOpen.test(rest) &&
            !comment.test(rest) &&
            !conditionalComment.test(rest)
          ) {
            // < in plain text, be forgiving and treat it as text
            next = rest.indexOf('<', 1);
            if (next < 0) { break }
            textEnd += next;
            rest = html.slice(textEnd);
          }
          text = html.substring(0, textEnd);
        }

        if (textEnd < 0) {
          text = html;
        }

        if (text) {
          advance(text.length);
        }

        if (options.chars && text) {
          options.chars(text, index - text.length, index);
        }
      } else {
        var endTagLength = 0;
        var stackedTag = lastTag.toLowerCase();
        var reStackedTag = reCache[stackedTag] || (reCache[stackedTag] = new RegExp('([\\s\\S]*?)(</' + stackedTag + '[^>]*>)', 'i'));
        var rest$1 = html.replace(reStackedTag, function (all, text, endTag) {
          endTagLength = endTag.length;
          if (!isPlainTextElement(stackedTag) && stackedTag !== 'noscript') {
            text = text
              .replace(/<!\--([\s\S]*?)-->/g, '$1') // #7298
              .replace(/<!\[CDATA\[([\s\S]*?)]]>/g, '$1');
          }
          if (shouldIgnoreFirstNewline(stackedTag, text)) {
            text = text.slice(1);
          }
          if (options.chars) {
            options.chars(text);
          }
          return ''
        });
        index += html.length - rest$1.length;
        html = rest$1;
        parseEndTag(stackedTag, index - endTagLength, index);
      }

      if (html === last) {
        options.chars && options.chars(html);
        if (!stack.length && options.warn) {
          options.warn(("Mal-formatted tag at end of template: \"" + html + "\""), { start: index + html.length });
        }
        break
      }
    }

    // Clean up any remaining tags
    parseEndTag();

    function advance (n) {
      index += n;
      html = html.substring(n);
    }

    function parseStartTag () {
      var start = html.match(startTagOpen);
      if (start) {
        var match = {
          tagName: start[1],
          attrs: [],
          start: index
        };
        advance(start[0].length);
        var end, attr;
        while (!(end = html.match(startTagClose)) && (attr = html.match(dynamicArgAttribute) || html.match(attribute))) {
          attr.start = index;
          advance(attr[0].length);
          attr.end = index;
          match.attrs.push(attr);
        }
        if (end) {
          match.unarySlash = end[1];
          advance(end[0].length);
          match.end = index;
          return match
        }
      }
    }

    function handleStartTag (match) {
      var tagName = match.tagName;
      var unarySlash = match.unarySlash;

      if (expectHTML) {
        if (lastTag === 'p' && isNonPhrasingTag(tagName)) {
          parseEndTag(lastTag);
        }
        if (canBeLeftOpenTag$$1(tagName) && lastTag === tagName) {
          parseEndTag(tagName);
        }
      }

      var unary = isUnaryTag$$1(tagName) || !!unarySlash;

      var l = match.attrs.length;
      var attrs = new Array(l);
      for (var i = 0; i < l; i++) {
        var args = match.attrs[i];
        var value = args[3] || args[4] || args[5] || '';
        var shouldDecodeNewlines = tagName === 'a' && args[1] === 'href'
          ? options.shouldDecodeNewlinesForHref
          : options.shouldDecodeNewlines;
        attrs[i] = {
          name: args[1],
          value: decodeAttr(value, shouldDecodeNewlines)
        };
        if (options.outputSourceRange) {
          attrs[i].start = args.start + args[0].match(/^\s*/).length;
          attrs[i].end = args.end;
        }
      }

      if (!unary) {
        stack.push({ tag: tagName, lowerCasedTag: tagName.toLowerCase(), attrs: attrs, start: match.start, end: match.end });
        lastTag = tagName;
      }

      if (options.start) {
        options.start(tagName, attrs, unary, match.start, match.end);
      }
    }

    function parseEndTag (tagName, start, end) {
      var pos, lowerCasedTagName;
      if (start == null) { start = index; }
      if (end == null) { end = index; }

      // Find the closest opened tag of the same type
      if (tagName) {
        lowerCasedTagName = tagName.toLowerCase();
        for (pos = stack.length - 1; pos >= 0; pos--) {
          if (stack[pos].lowerCasedTag === lowerCasedTagName) {
            break
          }
        }
      } else {
        // If no tag name is provided, clean shop
        pos = 0;
      }

      if (pos >= 0) {
        // Close all the open elements, up the stack
        for (var i = stack.length - 1; i >= pos; i--) {
          if (i > pos || !tagName &&
            options.warn
          ) {
            options.warn(
              ("tag <" + (stack[i].tag) + "> has no matching end tag."),
              { start: stack[i].start, end: stack[i].end }
            );
          }
          if (options.end) {
            options.end(stack[i].tag, start, end);
          }
        }

        // Remove the open elements from the stack
        stack.length = pos;
        lastTag = pos && stack[pos - 1].tag;
      } else if (lowerCasedTagName === 'br') {
        if (options.start) {
          options.start(tagName, [], true, start, end);
        }
      } else if (lowerCasedTagName === 'p') {
        if (options.start) {
          options.start(tagName, [], false, start, end);
        }
        if (options.end) {
          options.end(tagName, start, end);
        }
      }
    }
  }

# 优化器

在AST中找出静态子树并打上标记。标记静态子树有两点好处，

  1. 每次重新渲染时，不需要为静态子树创建新节点（VNode）；
  2. 在虚拟DOM中打补丁（patching）的过程可以跳过。

优化器内部实现主要分为两个步骤：

  1. 在AST中找出所有静态节点并打上标记；
  2. 在AST中找出所有静态根节点并打上标记。

静态节点的static属性为true，静态根节点的staticRoot属性也为true

//静态节点
{
    type:1,
    tag:'p',
    staticRoot:false,
    static:true,
    ......
}
    
//静态根节点
{
    type:1,
    tag:'p',
    staticRoot:true,
    static:true,
    ......
}

实现方式

export function optimize(root){
    if(!root) return;
    //找出静态节点并标记
    markStatic(root)
    //标记所有静态根节点
    markStaticRoots(root)
}

# 找出所有静态节点并标记

使用递归的方法， 先判断根节点是不是静态节点，接着判断其子节点，直到所有的节点被处理完毕。如果一个节点的子节点不是静态节点，那么这个节点一定不是静态节点。

function markStatic(node){
    node.static = isStatic(node)//判断节点是否为静态节点
    if(node.type === 1){
        //如果是元素节点，遍历其子节点并进行标记
        for(let i = 0,l = node.children.length;i<1;i++){
            const child = node.children[i]
            markStatic(child)
            if(!child.static){
                //如果子节点不是静态节点
                node.static = false
            }
        }
    }
}

//判断某个节点是否为静态节点
function isStatic(node){
    if(node.type === 2){
        //带变量的动态文本节点，不是静态节点
        return false
    }
    if(node.type === 3){
        //不带变量的动态文本节点，是静态节点
        return true
    }
    //如果是元素节点
    return !!(node.pre||	//没有使用v-pre指令
              (!node.hasBingdings &&	//没有动态绑定
               !node.if && !node.for &&	//没有v-for、v-if、v-else
               !isBuiltInTag(node.tag) &&)	//不是内置标签slot\component邓
               isPlatformReservedTag(node.tag) &&	//不是组件，必须是保留标签
              !isDirectChildOfTemplateFor(node) &&	//其父节点不能是带v-for的template标签
              Object.keys(node).every(isStaticKey)	//节点中不存在动态节点才有的属性
    ))
}

Object.keys(node).every(isStaticKey) ：静态节点的属性范围是type、tag、attrsList、attrsMap、plain、parent、children、attrs、staticClass、staticStyle。如果有不在这个范围内的属性那么他就不是静态节点。

# 找出所有静态根节点并标记

因为一个节点如果是静态节点，那么他的子节点也一定是静态节点。从上往下找，找到的第一个静态节点就是静态根节点。

有一种情况下，即便它是静态根节点也不会标记，因为其优化的成本大于收益。如下，p元素只有一个文本节点，成本大于收益。

<p>文本内容</p>

function markStaticRoots(node){
    if(node.type === 1){
       if(node.static && node.children.length &&
         !(node.children.length === 1 && node.children[0].type === 3)){
           node.staticRoot = true
           return
       }else{
           node.staticRoot = false
       }
       if(node.children){
           for(let i=0,l=node.children.length;i<l;i++){
               markStaticRoots(node.children[i])
           }
       }
    }
}

# 代码生成器

  代码生成器的作用是将AST转换成渲染函数中的内容，即代码字符串。代码生成器其实就是字符串拼接的过程。通过递归AST，从上向下来生成字符串。如：

<div id="el">Hello {{name}}</div>

  生成的代码字符串如下，其中_c是createElement函数的别称，用来生成元素节点，还有 _v是用来生成文本节点createTextVNode函数的别称， _e是createElementVNode生成注释节点函数的别称。 _c函数有三个参数，分别是元素标签名，元素属性数据对象，子节点列表。

`with(this){
	return _c(
		"div",
		{
			attrs:{'id':'el'}
		},
		[
			_v("Hello" + _s(name))
		]
	)
}`
